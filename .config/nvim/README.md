# starter
Starter config for NvChad

# help

- Lazy : plugins
- MasonInstall : install lsp server (autocompletion, errors)
- lspconfig.lua : use lsp server
- TSInstall : syntax highlighting
- conform.lua : lint

# vim

operators: d c p\
[number]\
motions: 0 b w|e $

replace: r R

new line: O|o

open previous telescope: alt n

next error: alt b
previous error: alt B

move half page: ctrl u|d
move screen : ctrl y/e

menu : ctrl y (tree or buf)
